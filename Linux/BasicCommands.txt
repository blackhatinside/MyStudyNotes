# extract a .tar.gz or .tar.bz2 file
tar xvzf file_name.tar.gz

# extract a tar.xz file
tar -xf file_name.tar.xz

# update GRUB after adding another entry into the booting menu (dual++ boots)
sudo update-grub

# know the current Desktop Environment used by Linux
echo $XDG_CURRENT_DESKTOP

# forcefully delete a directory/folder and its contents
sudo rm -rf folder_name

# to activate a Python virtual environmen
chmod +x venv/bin/activate
source venv/bin/activate
